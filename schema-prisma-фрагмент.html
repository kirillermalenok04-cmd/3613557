Фрагмент конфигурационного файла schema.prisma
описание моделей и реляционных связей

Файл: prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ─── Таблица 1: Пользователь ───
model User {
  id           Int      @id @default(autoincrement())
  email        String   @unique
  passwordHash String
  role         String   @default("customer")
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
  orders       Order[]  // связь 1:N → множество заказов
}

// ─── Таблица 2: Заказ (FK на User) ───
model Order {
  id        Int      @id @default(autoincrement())
  userId    Int?     // внешний ключ (nullable)
  status    String   @default("pending")
  payload   Json
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  user      User?    @relation(fields: [userId], references: [id], onDelete: SetNull)
  //           ↑ связь N:1 с User; при удалении User → userId = NULL
}

// ─── Таблица 3: Отзыв (независимая сущность) ───
model Review {
  id        Int      @id @default(autoincrement())
  payload   Json     // данные отзыва в JSON, без FK на User/Order
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}
Реляционные связи между таблицами:
  • User → Order (1 : N): поле orders Order[] в модели User и user User? + userId в модели Order; внешний ключ Order.userId ссылается на User.id.
  • Review: отдельная таблица без внешних ключей на User и Order; связь реализуется логически через JSON-поле payload (автор, текст, рейтинг и т.д.).
  • Итого в схеме: одна явная реляционная связь (User–Order); три таблицы в одной БД PostgreSQL (схема public).
