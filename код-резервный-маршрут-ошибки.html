Резервный маршрут и обработка ошибок на сервере Express

Файл: server.js

1. Резервный маршрут (SPA fallback)

Если запрос не совпал с API и статическим файлом, сервер отдаёт index.html — поддержка клиентской маршрутизации одностраничного приложения.

app.use(express.static(__dirname));

// SPA: всё остальное — index.html
app.get('*', (req, res) => {
    res.sendFile(path.join(__dirname, 'index.html'));
});

2. Обработка ошибок в маршрутах API (try / catch)

Асинхронные обработчики оборачиваются в try/catch: логирование в консоль и JSON-ответ с кодом состояния HTTP.

app.post('/api/auth/login', async (req, res) => {
    try {
        const { email, password } = req.body;
        const user = await prisma.user.findUnique({ where: { email } });

        if (!user) {
            return res.status(401).json({ error: 'Invalid credentials' });
        }

        const isValidPassword = await bcrypt.compare(password, user.passwordHash);
        if (!isValidPassword) {
            return res.status(401).json({ error: 'Invalid credentials' });
        }

        res.json({ user: { id: user.id, email: user.email, role: user.role } });
    } catch (error) {
        console.error('Login error:', error);
        res.status(500).json({ error: 'Login failed' });
    }
});

3. Обработка ошибки «ресурс не найден» (404)

При обновлении заказа перехватывается код Prisma P2025 — запись не найдена в БД.

app.put('/api/orders/:id', async (req, res) => {
    try {
        const { id } = req.params;
        const { status } = req.body;
        const order = await prisma.order.update({
            where: { id: Number(id) },
            data: { status }
        });
        res.json({ order: mapOrder(order), message: 'Order updated successfully' });
    } catch (error) {
        if (error.code === 'P2025') {
            return res.status(404).json({ error: 'Order not found' });
        }
        console.error('Error updating order:', error);
        res.status(500).json({ error: 'Failed to update order' });
    }
});

async function start() {
    try {
        await prisma.$connect();
        console.log('Database connected');
    } catch (error) {
        console.error('Database connection failed:', error);
        process.exit(1);
    }
    app.listen(PORT, '0.0.0.0', () => {
